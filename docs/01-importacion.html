  )) %>%

    date = update(fecha_de_siembra_dia_mes, year = 1),

                # sacar espacios antes-desp
                str_trim(
                  str_replace_all(., fixed(" "), "_")

  rowwise() %>%
  mutate(dano_tot = sum(dano_1,dano_2, na.rm = T)) %>%
